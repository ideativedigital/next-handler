Use `withApiHandler.enhance()` to resolve auth once per request and read it from AsyncLocalStorage in your handler code.

```ts
import { NextResponse } from "next/server";
import { withApiHandler, UnauthorizedError } from "@ideative/next-handler";

const validateAccessToken = async (token: string) => {
  if (token !== "demo-token") {
    throw new UnauthorizedError("Invalid access token");
  }
};

const authApiHandler = withApiHandler.enhance(async (req) => {
  const authHeader = req.headers.get("authorization");
  const token = authHeader?.startsWith("Bearer ")
    ? authHeader.slice("Bearer ".length)
    : null;

  if (!token) {
    throw new UnauthorizedError("Missing bearer token");
  }

  await validateAccessToken(token);
  return {
    user: {
      id: "u_123",
      token,
    },
  };
});

export const GET = authApiHandler(async (_req, context) => {
  const { user } = authApiHandler.context();
  return NextResponse.json({
    ok: true,
    userId: user.id,
    sameUserFromContext: context?.user.id,
  });
});
```

Enhancer callbacks can throw typed errors (like `UnauthorizedError`) and those still serialize through the same API error contract with `401` status responses.
