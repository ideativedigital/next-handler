Create an `EndpointError` subclass and register a deserializer:

```ts
import { EndpointError, apiErrorFactory } from "@ideative/next-handler";

class PaymentRequiredError extends EndpointError {
  static ErrorName() {
    return "PaymentRequiredError";
  }
  constructor(message = "Payment required") {
    super(PaymentRequiredError.ErrorName(), 402, message);
    this.name = PaymentRequiredError.ErrorName();
  }
}

apiErrorFactory.register(
  PaymentRequiredError,
  (d) => new PaymentRequiredError(d.message)
);
```
